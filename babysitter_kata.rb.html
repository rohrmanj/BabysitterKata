1 require 'pry'
 2 require 'Time'
 3 
 4 class BabySitter
 5   def initialize
 6     @start = Time.parse('5:00 PM')
 7     @bed_time = Time.parse('8:00 PM')
 8     @end = Time.parse('4:00 AM') + (24*60*60)
 9     @midnight = Time.parse('12:00 AM') + (24*60*60)
10 
11     @before_bed_rate = 12
12     @bed_to_midnight_rate = 8
13     @midnight_to_end_rate = 16
14   end
15 
16   def calc_pb_rate(start_time, bed_time, end_time)
17     parse_time(start_time, bed_time, end_time)
18     (((bed_time - start_time) / 3600) * @before_bed_rate).floor
19   end
20 
21   def calc_ab_rate(start_time, bed_time, end_time)
22     parse_time(start_time, bed_time, end_time)
23     bed_time < @midnight ? (((@midnight - bed_time) / 3600) * @bed_to_midnight_rate).floor : 0
24   end
25 
26   def calc_am_rate(start_time, bed_time, end_time)
27     parse_time(start_time, bed_time, end_time)
28     end_time > @midnight ? (((end_time - @midnight) / 3600) * @midnight_to_end_rate).floor : 0
29   end
30 
31   def parse_time(start_time, bed_time, end_time)
32     start_time = Time.parse(start_time)
33     bed_time = Time.parse(bed_time)
34     if end_time.end_with?("AM")
35       end_time = Time.parse(end_time) + (24*60*60)
36     elsif
37       end_time = Time.parse(end_time)
38     end
39 
40     return start_time, bed_time, end_time
41   end
42 
43   def calc_total
44     [calc_pb_rate, calc_ab_rate, calc_am_rate].sum
45   end
